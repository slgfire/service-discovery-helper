> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Architecture:** Add 4 new C source files (log.c/h, config.c/h) alongside existing code. All changes are additive — the core packet forwarding logic in `flood_packet()` is untouched. New features wire into existing globals. Dockerfile uses multi-stage build for multi-arch support.

**Tech Stack:** C99, GCC atomic builtins, libpcap, pthreads, syslog, Docker, GitHub Actions

**Spec:** `docs/superpowers/specs/2026-04-01-v2.0-design.md`

**Branch:** `v2.0` (already created)

**Note:** This is a C project with no test framework. Verification is done via `make` (compile check) and manual runtime testing where noted.

- [ ] **Step 2: Create `log.c`**

```c
#include "log.h"
#include <stdio.h>
#include <stdarg.h>
#include <time.h>
#include <syslog.h>

static int use_syslog_backend = 0;
int sdh_log_debug_enabled = 0;

void sdh_log_init(int use_syslog)
{
    use_syslog_backend = use_syslog;
    if (use_syslog)
        openlog("sdh-proxy", LOG_PID, LOG_DAEMON);
}

void sdh_log(int level, const char *fmt, ...)
{
    if (level == SDH_LOG_DEBUG && !sdh_log_debug_enabled)
        return;

    va_list args;
    va_start(args, fmt);

    if (use_syslog_backend)
    {
        int prio;
        switch (level)
        {
            case SDH_LOG_ERROR: prio = LOG_ERR; break;
            case SDH_LOG_DEBUG: prio = LOG_DEBUG; break;
            default:            prio = LOG_INFO; break;
        }
        vsyslog(prio, fmt, args);
    }
    else
    {
        time_t now = time(NULL);
        struct tm *t = localtime(&now);
        const char *prefix;
        switch (level)
        {
            case SDH_LOG_ERROR: prefix = "ERROR"; break;
            case SDH_LOG_DEBUG: prefix = "DEBUG"; break;
            default:            prefix = "INFO"; break;
        }
        fprintf(level == SDH_LOG_ERROR ? stderr : stdout,
                "[%04d-%02d-%02d %02d:%02d:%02d] %s: ",
                t->tm_year + 1900, t->tm_mon + 1, t->tm_mday,
                t->tm_hour, t->tm_min, t->tm_sec, prefix);
        vfprintf(level == SDH_LOG_ERROR ? stderr : stdout, fmt, args);
        fprintf(level == SDH_LOG_ERROR ? stderr : stdout, "\n");
    }

    va_end(args);
}

void sdh_log_cleanup(void)
{
    if (use_syslog_backend)
        closelog();
}
```

- [ ] **Step 2: Create `config.c`**

```c
#include "config.h"
#include "sdh-proxy.h"
#include "timer.h"
#include "log.h"
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <ctype.h>

extern char *iface_list[MAX_IFACES];
extern int num_ifaces;
extern int use_all_interfaces;
extern char *port_list[MAX_PORTS];
extern int num_ports;
extern int debug;
extern int logstat;
extern int sdh_log_debug_enabled;

enum config_section {
    SECTION_NONE,
    SECTION_INTERFACES,
    SECTION_PORTS,
    SECTION_SETTINGS
};

static char *strip_whitespace(char *str)
{
    while (*str && isspace(*str)) str++;
    char *end = str + strlen(str) - 1;
    while (end > str && isspace(*end)) *end-- = '\0';
    return str;
}

static int parse_setting(const char *key, const char *value)
{
    if (strcmp(key, "rate_limit") == 0)
    {
        timer_enabled = (strcmp(value, "yes") == 0) ? 1 : 0;
    }
    else if (strcmp(key, "rate_limit_timeout") == 0)
    {
        unsigned int ms = (unsigned int)atoi(value);
        if (ms == 0)
        {
            sdh_log(SDH_LOG_ERROR, "Invalid rate_limit_timeout: %s", value);
            return -1;
        }
        pkt_timeout_s = ms / 1000;
        pkt_timeout_us = (ms % 1000) * 1000;
        timer_enabled = 1;
    }
    else if (strcmp(key, "log_stats") == 0)
    {
        logstat = (strcmp(value, "yes") == 0) ? 1 : 0;
    }
    else if (strcmp(key, "syslog") == 0)
    {
        // Will be applied after config load in main()
        // Store in a global that main() reads
        if (strcmp(value, "yes") == 0)
            sdh_log_init(1);
    }
    else if (strcmp(key, "debug") == 0)
    {
        debug = (strcmp(value, "yes") == 0) ? 1 : 0;
        sdh_log_debug_enabled = debug;
    }
    else
    {
        sdh_log(SDH_LOG_ERROR, "Unknown setting: %s", key);
        return -1;
    }
    return 0;
}

int sdh_config_load(const char *path)
{
    FILE *fp = fopen(path, "r");
    if (!fp)
    {
        sdh_log(SDH_LOG_ERROR, "Cannot open config file: %s", path);
        return -1;
    }

    char line[256];
    enum config_section section = SECTION_NONE;

    while (fgets(line, sizeof(line), fp))
    {
        // Strip comments
        char *comment = strchr(line, '#');
        if (comment) *comment = '\0';

        char *trimmed = strip_whitespace(line);
        if (strlen(trimmed) == 0)
            continue;

        // Section headers
        if (trimmed[0] == '[')
        {
            if (strncmp(trimmed, "[interfaces]", 12) == 0)
                section = SECTION_INTERFACES;
            else if (strncmp(trimmed, "[ports]", 7) == 0)
                section = SECTION_PORTS;
            else if (strncmp(trimmed, "[settings]", 10) == 0)
                section = SECTION_SETTINGS;
            else
            {
                sdh_log(SDH_LOG_ERROR, "Unknown config section: %s", trimmed);
                fclose(fp);
                return -1;
            }
            continue;
        }

        switch (section)
        {
            case SECTION_INTERFACES:
                if (strcmp(trimmed, "auto") == 0)
                {
                    use_all_interfaces = 1;
                }
                else
                {
                    if (num_ifaces >= MAX_IFACES)
                    {
                        sdh_log(SDH_LOG_ERROR, "Too many interfaces (max %d)", MAX_IFACES);
                        fclose(fp);
                        return -1;
                    }
                    iface_list[num_ifaces] = malloc(strlen(trimmed) + 1);
                    strcpy(iface_list[num_ifaces], trimmed);
                    num_ifaces++;
                }
                break;

            case SECTION_PORTS:
                if (num_ports >= MAX_PORTS)
                {
                    sdh_log(SDH_LOG_ERROR, "Too many ports (max %d)", MAX_PORTS);
                    fclose(fp);
                    return -1;
                }
                port_list[num_ports] = malloc(strlen(trimmed) + 1);
                strcpy(port_list[num_ports], trimmed);
                num_ports++;
                break;

            case SECTION_SETTINGS:
            {
                char *eq = strchr(trimmed, '=');
                if (!eq)
                {
                    sdh_log(SDH_LOG_ERROR, "Invalid setting (missing '='): %s", trimmed);
                    fclose(fp);
                    return -1;
                }
                *eq = '\0';
                char *key = strip_whitespace(trimmed);
                char *value = strip_whitespace(eq + 1);
                if (parse_setting(key, value) != 0)
                {
                    fclose(fp);
                    return -1;
                }
                break;
            }

            case SECTION_NONE:
                sdh_log(SDH_LOG_ERROR, "Config entry outside of section: %s", trimmed);
                fclose(fp);
                return -1;
        }
    }

    fclose(fp);
    sdh_log(SDH_LOG_INFO, "Loaded config from %s", path);
    return 0;
}
```
